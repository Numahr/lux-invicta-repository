# Family member marries childhood friend (AI)
character_event = {
	id = VIETmisc.3100
	hide_window = yes
	
	ai = yes
	capable_only = yes
	prisoner = no
	min_age = 18
	max_age = 30
	only_men = yes
	
	trigger = {
		is_married = no
		is_betrothed = no
		is_primary_heir = no
		is_pretender = no
		NOT = { trait = chaste }
		liege = {
			NOT = { any_child = { character = ROOT } }
			is_patrician = yes
			dynasty = ROOT
		}
	}
	
	mean_time_to_happen = {
		months = 250
	}
	
	immediate = {
		random_list = {
			33 = {
				liege = {
					character_event = { id = VIETmisc.3101 }
				}
			}
			33 = {
				liege = {
					character_event = { id = VIETmisc.3102 }
				}
			}
			33 = {
				liege = {
					character_event = { id = VIETmisc.3103 }
				}
			}
		}
	}
	
	option = {
		name = OK
	}
}
