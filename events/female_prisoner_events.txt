# The prisoner is released into the bedchamber
character_event = {
	id = MyMod.802
	desc = EVTDESCMyMod.802
	
	is_friendly = yes
	
	picture = GFX_evt_bed
	
	is_triggered_only = yes
	
	immediate = {
		prisoner = no
	}
	
	option = {
		name = EVTOPTAMyMod.802 # Free at last!
		FROM = {
			remove_trait = idealistic
			remove_trait = noblespirit
			remove_trait = gullible
			remove_trait = aloof
		
			
			reverse_opinion = {
				who = ROOT
				modifier = opinion_sprung_from_jail
				months = 240
			}
			if = {
				limit = { 
					fertility = 0.1
					ROOT = {
						fertility = 0.1
					}
					is_married = yes
				}
				if = {
					limit = {
						NOT = {
							trait = deceitful
						}
					}
					random = {
						chance = 30 
						add_trait = deceitful
					}
				}
				random = {
					chance = 40
					modifier = {
						factor = 1.1
						fertility = 0.3
					}
					modifier = {
						factor = 1.1
						fertility = 0.5
					}
					modifier = {
						factor = 1.1
						fertility = 0.7
					}
					modifier = {
						factor = 1.1
						ROOT = {
							fertility = 0.3
						}
					}
					modifier = {
						factor = 1.1
						ROOT = {
							fertility = 0.5
						}
					}
					modifier = {
						factor = 1.1
						ROOT = {
							fertility = 0.7
						}
					}
					ROOT = {
						impregnate_cuckoo = PREV
					}
				}
				set_character_flag = love_mistress
			}
			if = {
				limit = { 
					fertility = 0.1
					ROOT = {
						fertility = 0.1
					}
					is_married = no
				}
				
				random = {
					chance = 40
					ROOT = { impregnate = PREV }
				}
				set_character_flag = love_mistress
			}
			random = {
				chance = 25
				add_lover = ROOT
			} 
		}
	}
}
