character_event = { # Battlefield Duel Start
    id = duel_be.0
    desc = "EVTDESC5556000"
    border = "GFX_event_normal_frame_war"
    picture = GFX_evt_battle
    
    is_triggered_only = yes
    #triggered by on_combat_pulse
    
 	
    
    weight_multiplier = { #on_action, multiplies with chance there
        days = 1
        modifier = {
            factor = 1.5
            trait = trained_warrior
        }
        modifier = {
            factor = 1.75
            trait = skilled_warrior
        }
        modifier = {
            factor = 2
            trait = master_warrior
        }
        modifier = {
            factor = 3
            trait = duelist
        }
        modifier = {
            factor = 1.5
            trait = brave
        }
        modifier = {
            factor = 0.5
            trait = craven
        }
    }
   trigger = {
		# Avoid crossing targets
       FROM = {
            liege = {
                OR = {
                    any_realm_character = {
                        in_battle = yes
                        at_location = ROOT
                        NOT = { has_character_modifier = battlefield_fight }
						# Restrict to Heroic cultures and Martial religions (glory in battle)
						OR = {
							heroic_culture_trigger = yes
							martial_religion_trigger = yes
						}
                   }
					AND = {
						in_battle = yes
						at_location = ROOT
						NOT = { has_character_modifier = battlefield_fight }
						# Restrict to Heroic cultures
						OR = {
							heroic_culture_trigger = yes
							martial_religion_trigger = yes
						}
					}
                }
            }
        }
		OR = {
			heroic_culture_trigger = yes
			martial_religion_trigger = yes
		}
		NOR = {
			has_character_modifier = battlefield_fight
			trait = duel_target
		}
	}
	
	immediate = {
        # Target lock
		add_character_modifier = {
			name = battlefield_fight
			duration = 20
		}
        FROM = {
            liege = {
                if = {
                    limit = {
                        NOR = {
                            in_battle = yes
                            at_location = ROOT
                            NOT = { has_character_modifier = battlefield_fight }
							NOT = { trait = duel_target }
                        }
						any_realm_character = {
							in_battle = yes
							at_location = ROOT
							prisoner = no
							is_alive = yes		
							NOT = { has_character_modifier = battlefield_fight }
							NOT = { trait = duel_target }
						}
					}
					random_realm_character = {
                        limit = {
                            in_battle = yes
                            at_location = ROOT
                            NOT = { has_character_modifier = battlefield_fight }
							NOT = { trait = duel_target }
							OR = {
								has_job_title = job_marshal
								trait = peasant_leader
								trait = heresiarch
								AND = {
									is_ruler = no
									ROOT = {
										OR = {
											is_ruler = no
											primary_title = { lower_tier_than = COUNT }
											has_job_title = job_marshal
											trait = peasant_leader
											trait = heresiarch
										}
									}
								}
								AND = {
									primary_title = { higher_tier_than = BARON }
									ROOT = {
										OR = {
											primary_title = { higher_tier_than = BARON }
											has_job_title = job_marshal
											trait = peasant_leader
											trait = heresiarch
										}
									}
								}
							}
                        }
                        opinion = { who = ROOT modifier = opinion_battle_duel_target }
                        reverse_opinion = { who = ROOT modifier = opinion_battle_duel_target }
                        add_character_modifier = {
                            name = battlefield_fight
                            duration = 20
                        }
                    }
                    ROOT = { set_character_flag = duel_engaged }
                }
                if = {
                    limit = {
                        NOT = {
                            any_realm_character = {
                                in_battle = yes
                                at_location = ROOT
                                NOT = { has_character_modifier = battlefield_fight }
								NOT = { trait = duel_target }
                            }
                        }
						in_battle = yes
						at_location = ROOT
						NOT = { has_character_modifier = battlefield_fight }
						NOT = { trait = duel_target }
                    }
                    opinion = { who = ROOT modifier = opinion_battle_duel_target }
                    reverse_opinion = { who = ROOT modifier = opinion_battle_duel_target }
                    add_character_modifier = {
                        name = battlefield_fight
                        duration = 20
                    }
                    ROOT = { set_character_flag = duel_engaged }
                }
                if = {
                    limit = {
						NOT = { ROOT = { has_character_flag = duel_engaged } }
						any_realm_character = {
							in_battle = yes
							at_location = ROOT
							prisoner = no
							is_alive = yes		
							NOT = { has_character_modifier = battlefield_fight }
							NOT = { trait = duel_target }
						}
						in_battle = yes
						at_location = ROOT
						prisoner = no
						is_alive = yes		
						NOT = { has_character_modifier = battlefield_fight }
						NOT = { trait = duel_target }
					}
                    random_list = {
                        10 = {
							opinion = { who = ROOT modifier = opinion_battle_duel_target }
							reverse_opinion = { who = ROOT modifier = opinion_battle_duel_target }
                            add_character_modifier = {
                                name = battlefield_fight
                                duration = 20
                            }
                        }
                        90 = {
                            random_realm_character = {
                                limit = {
                                    in_battle = yes
                                    at_location = ROOT
                                    NOT = { has_character_modifier = battlefield_fight }
									NOT = { trait = duel_target }
                                }
                                opinion = { who = ROOT modifier = opinion_battle_duel_target }
								reverse_opinion = { who = ROOT modifier = opinion_battle_duel_target }
                                add_character_modifier = {
                                    name = battlefield_fight
                                    duration = 20
                                }
                            }
                        }
                    }
                }
            }
        }
        clr_character_flag = duel_engaged
    }
	
	option = {
		name = "EVTOPTA5556000"
        FROM = {
            liege = {
                random_realm_character = {
                    limit = { has_opinion_modifier = { who = ROOT modifier = opinion_battle_duel_target } }
					hidden_tooltip = { character_event = { id = duel_be.1 } }
                }
                if = {
                    limit = { has_opinion_modifier = { who = ROOT modifier = opinion_battle_duel_target } }
                    hidden_tooltip = { character_event = { id = duel_be.1 } }
                }
            }
        }
    }
}
