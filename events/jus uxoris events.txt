#on_death event for usurping husband - no sons by usurped wife
character_event = {
	id = jus_uxoris.1
	desc = EVTDESCju.1
	picture = GFX_evt_death
	
	is_triggered_only = yes
	
	trigger = {
		has_character_flag = usurped_my_wife
		any_spouse = {
			has_character_flag = usurped_by_husband
			NOT = {
				any_child = {
					father = {
						character = ROOT
					}
					is_female = no
					is_alive = yes
				}
			}
			any_claim = {
				has_holder = yes
				holder_scope = {
					character = ROOT
				}
			}
		}
	}
	
	immediate = {
		clr_character_flag = usurped_my_wife
		primary_title = {
			ROOT = {
				any_spouse = {
					limit = {
						has_character_flag = usurped_by_husband
					}
					clr_character_flag = usurped_by_husband
					primary_title = {
						PREV = {
							any_claim = {
								limit = {
									has_holder = yes
									holder_scope = {
										character = ROOT
									}
									PREV = {
										has_strong_claim = PREV
									}
								}
								ROOT = {
									if = {
										limit = {
											NOT = {
												primary_title = {
													title = PREVPREV
												}
											}
										}
										PREV = {
											gain_title_plus_barony_if_unlanded = PREVPREVPREV
										}
									}
									if = {
										limit = {
											primary_title = {
												title = PREVPREV
											}
										}
										PREV = {
											gain_title_plus_barony_if_unlanded = PREVPREVPREV
										}
										if = {
											limit = {
												PREV = {
													is_ruler = yes
												}
											}
											primary_title = {
												copy_title_laws = PREVPREV
											}
										}
									}
								}
								copy_title_history = PREVPREV
							}
							character_event = {
								id = jus_uxoris.2		#Notify wife
							}
						}
					}
					ROOT = {
						any_vassal = {
							limit = {
								has_character_flag = usurped_by_liege_husband
							}
							set_defacto_liege = PREVPREV
							clr_character_flag = usurped_by_liege_husband
							character_event = {
								id = jus_uxoris.3	#Notify vassal
							}
						}
						if = {
							limit = {
								has_character_flag = vassal_husband
							}
							set_defacto_liege = PREV
						}
					}
					any_vassal = {
						limit = {
							has_character_flag = usurped_by_liege_husband
						}
						clr_character_flag = usurped_by_liege_husband
						character_event = {
							id = jus_uxoris.3		#Notify vassal
						}
					}
				}
			}
		}
	}
	option = {
		name = ALAS
	}
}

character_event = {
	id = jus_uxoris.21
	desc = EVTDESCju.21
	picture = GFX_evt_quarrel
	
	is_triggered_only = yes
	
	option = {
		name = PRAISEGOD
		clr_character_flag = usurped_by_husband
		custom_tooltip = {
			text = CUSTOMTOOLTIPju.22
		}
		primary_title = {
			FROM = {
				primary_title = {
					ROOT = {
						any_claim = {
							limit = {
								has_holder = yes
								holder_scope = {
									character = FROM
								}
								ROOT = {
									has_strong_claim = PREV
								}
							}
							PREVPREV = {
								if = {
									limit = {
										NOT = {
											title = PREV
										}
									}
									gain_title_plus_barony_if_unlanded = ROOT
								}
								if = {
									limit = {
										title = PREV
									}
									gain_title_plus_barony_if_unlanded = ROOT
									copy_title_laws = PREV
								}
							}
						}
						FROM = {
							any_vassal = {
								limit = {
									has_character_flag = usurped_by_liege_husband
								}
								clr_character_flag = usurped_by_liege_husband
								set_defacto_liege = ROOT
								hidden_tooltip = {
									character_event = {
										id = jus_uxoris.22	#Notify vassal
									}
								}
							}
							if = {
								limit = {
									has_character_flag = vassal_husband
								}
								set_defacto_liege = ROOT
							}
						}
						any_vassal = {
							limit = {
								has_character_flag = usurped_by_liege_husband
							}
							clr_character_flag = usurped_by_liege_husband

							character_event = {
								id = jus_uxoris.22	#Notify vassal
							}
						}
					}
				}
			}
		}
	}
}
