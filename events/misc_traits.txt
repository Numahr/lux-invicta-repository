#misc_traits.1 - Child likes dresses. (Fashionable)
#on_yearly_childhood_pulse (20)
character_event = {
	id = misc_traits.1
	desc = "EVTDESCmisc_traits.1"
	picture = "GFX_fashion"
	
	min_age = 6
	max_age = 16
	
	is_triggered_only = yes
	
	trigger = {
		has_guardian = yes
		NOR = {
			trait = shy
			trait = humble
		}
	}
	
	weight_multiplier = {
		days = 1
		
		#Pro
		modifier = {
			factor = 2
			is_female = yes
		}
		modifier = {
			factor = 1.5
			OR = {
				trait = fair
				trait = envious
				trait = proud
				trait = gregarious
			}
		}
	
		#Con
		modifier = {
			factor = 0.5
			OR = {
				trait = ugly
				trait = dwarf
				trait = stutter
				trait = content
			}
		}
	}
	
	option = {
		name = "EVTOPTAmisc_traits.1"
		guardian = {
			character_event = {
				id = misc_traits.2
				tooltip = "EVTOOLTIPmisc_traits.1" #Guardian will descide.
			}
		}
	}
}

#misc_traits.2 - My ward likes dresses. (Fashionable)
character_event = {
	id = misc_traits.2
	desc = "EVTDESCmisc_traits.2"
	picture = "GFX_fashion"
	
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAmisc_traits.2" #Encourage him.
		
		ai_chance = {
			factor = 1
			modifier = {
				factor = 5
				OR = {
					trait = gregarious
					trait = lustful
					trait = poet
					trait = fashion
					trait = hedonist
				}
			}
		}
		
		if = {
			limit = {
				OR = {
					trait = gregarious
					trait = lustful
					trait = poet
					trait = fashion
					trait = hedonist
				}
			}
			random_list = {
				50 = { change_diplomacy = 1 }
				50 = { }
			}
		}
		FROM ={
			add_trait = fashion
			character_event = {
				id = misc_traits.3
				tooltip = "EVTOOLTIPmisc_traits.3" #Ward gets fashionable.
			}
		}
	}
	
	option = {
		name = "EVTOPTBmisc_traits.2" #Tell him to stop it!
		ai_chance = {
			factor = 1
			modifier = {
				factor = 5
				OR = {
					trait = shy
					trait = wroth
					trait = celibate
					trait = hunchback
				}
			}
		}
		FROM ={
			character_event = {
				id = misc_traits.4
				tooltip = "EVTOOLTIPmisc_traits.4" #Ward no gets fashionable.
			}
		}
	}
