# cabbages are too lowly
character_event = {
	id = VIETmisc.2002
	desc = VIETmisc.2002.desc
	picture = GFX_evt_courtiers_talking
	
	is_triggered_only = yes
	
	min_age = 16
	only_rulers = yes
	capable_only = yes
	prisoner = no
	war = no
	
	trigger = {
		in_command = no
		is_abroad = no
		any_courtier = {
			NOT = { trait = incapable }
			prisoner = no
			age = 16
			in_command = no
			is_abroad = no
			OR = {
				trait = proud
				trait = gluttonous
				trait = arbitrary
			}
			NOT = { trait = humble }
			NOT = { trait = temperate }
		}
	}
	weight_multiplier = {
			days = 1
			modifier = {
					factor = 1.2
					trait = gregarious
			}
			modifier = {
					factor = 0.8
					trait = shy
			}
			modifier = {
					factor = 1.2
					trait = proud
			}
			modifier = {
					factor = 0.8
					trait = humble
			}
			modifier = {
					factor = 1.2
					trait = ambitious
			}
			modifier = {
					factor = 0.8
					trait = content
			}
			modifier = {
					factor = 1.2
					trait = paranoid
			}
			modifier = {
					factor = 0.8
					trait = trusting
			}
			modifier = {
					factor = 1.2
					trait = just
			}
			modifier = {
					factor = 0.8
					trait = arbitrary
			}
			modifier = {
					factor = 1.2
					trait = kind
			}
			modifier = {
					factor = 0.8
					trait = cruel
			}
	}
	
	immediate = {
		random_courtier = {
			limit = {
				NOT = { trait = incapable }
				prisoner = no
				age = 16
				in_command = no
				is_abroad = no
				OR = {
					trait = proud
					trait = gluttonous
					trait = arbitrary
				}
				NOT = { trait = humble }
				NOT = { trait = temperate }
			}
			save_event_target_as = target_VIET_cabbage_hater
		}		
	}
	
	option = { # agree
		name = VIETmisc.2002.a
		if = {
			limit = {
				trait = humble
			}
			random = {
				remove_trait = humble
				chance = 40
			}
		}
		else = {
			random = {
				add_trait = proud
				chance = 25
			}
		}
		event_target:target_VIET_cabbage_hater = {
			opinion = {
					modifier = opinion_satisfied
					who = ROOT
					years = 1
			}
		}
		if = {
			limit = {
				religion_group = muslim
			}
			decadence = 1
		}
	}
	option = { # disagree
		name = VIETmisc.2002.b
		if = {
			limit = {
				trait = proud
			}
			random = {
				remove_trait = proud
				chance = 25
			}
		}
		else = {
			random = {
				add_trait = humble
				chance = 20
			}
		}
		event_target:target_VIET_cabbage_hater = {
			opinion = {
					modifier = opinion_disappointed
					who = ROOT
					years = 1
			}
		}
	}
	option = { # make a joke
		name = VIETmisc.2002.c
		tooltip_info = diplomacy
		trigger = {
			diplomacy = 8
		}
		prestige = 3
	}
}

	id = VIETmisc.2021
	desc = VIETmisc.2021.desc

			religion_group = christian

			AND = {
				religion_group = indian_group
				OR = {
					trait = zealous
					AND  = {
						trait = cruel
						NOT = { trait = cynical }
					}
				}
			}
			AND = {
				religion_group = pagan_group
				OR = {
					trait = zealous
					AND  = {
						trait = cruel
						NOT = { trait = cynical }
					}
					has_religion_feature = religion_warlike
					has_religion_feature = religion_dogmatic
				}
				NOT = { has_religion_feature = religion_peaceful }
				NOT = { has_religion_feature = religion_syncretic }
			}
			#religion = buddhist
			#religion = jain
			#religion = nestorian
			#religion = priscillianism
			#religion = qarmatian
			#religion = shiite
			#religion = african_solar_pagan
			#religion = african_solar_pagan_reformed
			#religion = latin_solar_pagan
			#religion = latin_solar_pagan_reformed
			#religion = mazdakism
			religion = tao
		}
		NOT = { has_alternate_start_parameter = { key = religion_names value = random } }
		#NOT = { has_alternate_start_parameter = { key = culture_names value = random } }
	}
	
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 1.2
			trait = diligent
		}
		modifier = {
			factor = 0.8
			trait = slothful
		}
		modifier = {
			factor = 1.2
			trait = impaler
		}
		modifier = {
			factor = 0.8
			trait = scholar
		}
		modifier = {
			factor = 1.2
			trait = just
		}
		modifier = {
			factor = 0.8
			trait = arbitrary
		}
		modifier = {
			factor = 1.2
			trait = zealous
		}
		modifier = {
			factor = 0.8
			trait = cynical
		}
		modifier = {
			factor = 1.2
			trait = wroth
		}
		modifier = {
			factor = 0.8
			trait = patient
		}
		modifier = {
			factor = 1.2
			trait = ambitious
		}
		modifier = {
			factor = 0.8
			trait = content
		}
		modifier = {
			factor = 1.2
			martial = 15
		}
		modifier = {
			factor = 0.8
			NOT = { martial = 6 }
		}
	}
	
	option = { #Kill Kill Kill
		name = VIETmisc.2021.a
		piety = 3
	}
	option = { #turn the other cheek
		name = VIETmisc.2021.b
		random_list = {
			50 = { piety = 5 }
			50 = { piety = -5 }
		}
		random = {
			add_trait = kind
			chance = 25
		}
	}
	option = { #dont be a hypocrite
		name = VIETmisc.2021.c
		piety = -2
		prestige = 4
		if = {
			limit = {
				trait = zealous
			}
			random = {
				remove_trait = zealous
				chance = 40
			}
		}
	}
}

# A bad bad vassal
character_event = {
	id = VIETmisc.2023
	desc = VIETmisc.2023.desc
	picture = GFX_evt_emissary
	
	is_triggered_only = yes
	
	min_age = 16
	only_rulers = yes
	capable_only = yes
	prisoner = no
	war = no
	
	trigger = {
		in_command = no
		is_abroad = no
		OR = {
			religion = antinomian
			religion = bogomilist
			religion = cathar
			religion = fraticelli
			religion = kharijite
			religion = mazdakism
			religion = tao
			religion = waldensian
		}
		any_vassal = {
			is_adult = yes
			is_feudal = yes
			religion = ROOT
			wealth = 100
			ai = yes
			NOT = { trait = incapable }
			NOT = { trait = charitable }
			NOT = { trait = temperate }
			NOT = { trait = kind }
			NOT = { trait = just }
		}
		NOT = { has_alternate_start_parameter = { key = religion_names value = random } }
		#NOT = { has_alternate_start_parameter = { key = culture_names value = random } }
	}
	
	weight_multiplier = {
		days = 1
		modifier = {
			factor = 1.2
			trait = gregarious
		}
		modifier = {
			factor = 0.8
			trait = shy
		}
		modifier = {
			factor = 1.2
			trait = charitable
		}
		modifier = {
			factor = 0.8
			trait = greedy
		}
		modifier = {
			factor = 1.2
			trait = just
		}
		modifier = {
			factor = 0.8
			trait = arbitrary
		}
		modifier = {
			factor = 1.2
			trait = patient
		}
		modifier = {
			factor = 0.8
			trait = wroth
		}
		modifier = {
			factor = 1.2
			trait = kind
		}
		modifier = {
			factor = 0.8
			trait = cruel
		}
	}
	
	immediate = {
		random_vassal = {
			limit = {
				is_adult = yes
				is_feudal = yes
				religion = ROOT
				wealth = 100
				ai = yes
				NOT = { trait = incapable }
				NOT = { trait = charitable }
				NOT = { trait = temperate }
				NOT = { trait = kind }
				NOT = { trait = just }
			}
			save_event_target_as = target_VIET_bad_religion_greed_vassal
		}	
	}
	
	option = { #give it back to the ppl
		name = VIETmisc.2023.a
		piety = 5
		event_target:target_VIET_bad_religion_greed_vassal = {
			wealth = -50
			opinion = {
				modifier = opinion_angry
				who = ROOT
				years = 5
			}
		}
		
	}
	option = { #its an ideal
		name = VIETmisc.2023.b
		piety = -10
		event_target:target_VIET_bad_religion_greed_vassal = {
			opinion = {
				modifier = opinion_happy
				who = ROOT
				years = 1
			}
		}
	}
}
