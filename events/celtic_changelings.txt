	id = ancrel.214
	hide_window = yes
	
	only_playable = yes
	only_men = yes

		is_married = yes
		spouse = {
			is_pregnant = yes
			NOT = {
				has_character_flag = child_stolen_sidhe

			trait = druid
		}

			name = supernatural_events

	}

		spouse = {
			character_event = {

		}
	}
	
}
