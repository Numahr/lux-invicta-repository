#Ambush Volley#
sk_as_ambushvolley = {
	days = 4
	sprite = 6
	group = sk_archerystrikes

	trigger = {
		phase = skirmish
		flank_has_leader = yes #required
		archers = 0.2	#left-most bell-curve trigger
		leader = {
			intrigue = 8
		}
		OR = {
			AND = {
				location = {
				OR = {
				terrain = forest
				terrain = hills				
				}
			}
				leader = {
				trait = rough_terrain_leader
				}
			}
			AND = {
				location = {
				terrain = mountain
			}
				leader = {
				trait =  mountain_terrain_leader
				}
			}
			AND = {
				location = {
				terrain = jungle
			}
				leader = {
				trait = jungle_terrain_leader
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 3	
		# commander -- redundant bonus to make them trigger more likely
		modifier = {
			factor = 5
			flank_has_leader = yes #presence bonus
		}
		modifier = {
			factor = 5
			leader = {
				OR = {
				trait = rough_terrain_leader
				trait =  mountain_terrain_leader
				trait = jungle_terrain_leader				
				}
			}
		}
		#bell-curve
		modifier = {
			factor = 1.5
			archers = 0.25
		}
		modifier = {
			factor = 3
			archers = 0.3
		}
		#location  unimplemented
	}
	
	archers_offensive = 1.5
	archers_defensive = 1
	light_infantry_offensive = 1
	
	enemy = {
		group = sk_manoeuvres
		factor = -0.25
	}
	enemy = {
		group = sk_cavalrystrikes
		factor = 0.75
	}
	enemy = {
		group = sk_infantryscreening
		factor = 0.75
	}
}
